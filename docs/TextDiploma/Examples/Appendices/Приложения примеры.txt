Серверное приложение – это программное обеспечение, которое работает на сервере и предоставляет различные сервисы или функциональность для клиентских приложений. Серверные приложения обычно используются в клиент-серверной архитектуре. 
API (Application Programming Interface) – это набор правил и протоколов, которые позволяют различным программным приложениям взаимодействовать друг с другом. 
REST (Representational State Transfer) – это стиль архитектуры программного обеспечения для распределенных систем, который, как правило, используется для построения веб-служб. 
Слой представления данных – это первый и самый верхний уровень, который присутствует в приложении. Этот уровень представляет собой UI, то есть представление контента конечному пользователю, через интерфейс. 
HTTP-запрос – это запрос, который отправляется от клиента к веб-серверу для получения или отправки данных. 
Контроллер – это компонент приложения, который обрабатывает входящие запросы от пользователей или других систем. 
Сервис – это слой бизнес-логики приложения. 
Бизнес-логика – это набор правил, необходимых для запуска приложения в соответствии с руководящими принципами, установленными организацией. 
Репозиторий – это компонент, отвечающий за взаимодействие с хранилищем данных.

    id          serial primary key,

    cart_id         int references cart (id),
    product_item_id int,
    quantity        int,
    is_selected     bool not null default false
);

create table if not exists client
(
    id      serial primary key,
    name    text not null default '',
    inn     text,
    address text
);

create table if not exists orders
(
    id            uuid primary key,
    status        text      not null,
    create_date   timestamp not null,
    delivery_date timestamp,
    user_id       int
    );

create table if not exists orders_item
(
    id              uuid primary key,
    orders_id       uuid references orders (id),
    product_item_id int,
    quantity        int
    );

create table if not exists operation
(
    id          uuid primary key,
    date        timestamp not null,
    description text,
    orders_id   uuid references orders (id)
    );

create table if not exists operation_detail
(
    id           uuid primary key,
    operation_id uuid references operation (id),

(
    id          serial primary key,
    status      text not null default 0,
    type        text not null default 0,

(
    id              serial primary key,
    request_id      int,
    product_id      int,
    product_item_id int,
    product_item_count int

(
    id   serial primary key,

create table if not exists users
(
    id        serial primary key,
    username  text not null default '',
    password  text not null default '',
    status    int,
    role_id   int references user_role (id),
    client_id int
);

create table if not exists product
(
    id          serial primary key,
    name        text    default ''::text not null,
    brand       text    default ''::text not null,
    price       numeric default 0.0      not null,
    material    text    default ''::text not null,
    status      text    default ''::text not null,
    factory_id  integer
);

create table if not exists product_item
(
    id         serial primary key,
    product_id integer references product(id),
    color      text    default ''::text not null,
    size       numeric default 0.0      not null,
    weight     numeric default 0.0      not null,
    quantity   integer default 0        not null
);

create table if not exists material
(
    id   serial primary key,
    name text not null
);

create table if not exists product_material
(
    id          serial  primary key,
    product_id  integer references product(id),
    material_id integer references material(id)
);

create table if not exists image
(
    id         serial primary key,
    product_id integer references product(id),
    url        text default ''::text not null
);

select pit.id       as id,
       p.id         as product_id,
       p.factory_id as factory_id,
       p.name       as name,
       p.price      as price,
       pit.color    as color,
       pit.size     as size,
       pit.weight   as weight,
       pit.quantity as count
from product_item pit
    join product p on p.id = pit.product_id
where pit.id = any ($1)

func TestImpl_UpdateStatus(t *testing.T) {
	ctx := goatcontext.NewContext()
	requestId := 123
	productId := "product-1"

	testProduct := domain.Product{
		Id:   productId,
		Name: "Test Product",
		Items: []domain.ProductItem{
			{
				Id:    productId,
				Color: "Red",
				Size:  "M",
			},
		},
	}

	testRequest := database.Request{
		Id:         requestId,
		Status:     "PENDING",
		Type:       "SUPPLY",
		UpdateDate: time.Now(),
		Items: []database.RequestItem{
			{
				ProductId:        productId,
				ProductItemCount: 5,
			},
		},
	}

	t.Run("успешное обновление статуса (не ApprovedStatus)", func(t *testing.T) {
		mockRepo := new(MockRequestRepository)
		mockWarehouse := new(MockWarehouseClient)
		mockNotifier := new(MockNotifierClient)

		service := NewRequestService(mockRepo, &warehouse.Client{Client: mockWarehouse}, &notifier.Client{Client: mockNotifier})

		mockRepo.On("UpdateRequestStatus", ctx, requestId, "REJECTED").Return(nil)

		err := service.UpdateStatus(ctx, requestId, domain.Status("REJECTED"))

		assert.NoError(t, err)
		mockRepo.AssertExpectations(t)
		mockWarehouse.AssertNotCalled(t, "GetDetailedProduct")
		mockNotifier.AssertNotCalled(t, "SendMail")
	})

	t.Run("успешное обновление статуса до ApprovedStatus", func(t *testing.T) {
		mockRepo := new(MockRequestRepository)
		mockWarehouse := new(MockWarehouseClient)
		mockNotifier := new(MockNotifierClient)

		service := NewRequestService(mockRepo, &warehouse.Client{Client: mockWarehouse}, &notifier.Client{Client: mockNotifier})

		mockRepo.On("UpdateRequestStatus", ctx, requestId, "APPROVED").Return(nil)
		mockRepo.On("GetRequestById", ctx, requestId).Return(testRequest, nil)
		mockWarehouse.On("GetDetailedProduct", ctx, productId).Return(testProduct, nil)
		mockNotifier.On("SendMail", ctx, mock.Anything).Return(nil)

		err := service.UpdateStatus(ctx, requestId, domain.ApprovedStatus)

		assert.NoError(t, err)
		mockRepo.AssertExpectations(t)

		time.Sleep(100 * time.Millisecond)
		mockWarehouse.AssertExpectations(t)
		mockNotifier.AssertExpectations(t)
	})

	t.Run("ошибка при обновлении статуса", func(t *testing.T) {

		mockNotifier := new(MockNotifierClient)

		assert.Error(t, err)
		assert.Equal(t, expectedErr, err)
		mockRepo.AssertExpectations(t)
		mockWarehouse.AssertNotCalled(t, "GetDetailedProduct")
		mockNotifier.AssertNotCalled(t, "SendMail")
	})
	t.Run("ошибка при отправке сообщения после обновления до ApprovedStatus", func(t *testing.T) {

БД (База данных) – это имеющая название совокупность данных, которая отражает состояние объектов и их отношений в рассматриваемой предметной области.
СУБД (Система управления базами данных) – это совокупность программных и лингвистических средств общего или специального назначения, обеспечивающих управление созданием и использованием баз данных.
Microsoft SQL Server (MS SQL Server) – это реляционная система управления базами данных (СУБД) от компании Microsoft, предназначена для хранения, обработки и анализа информации в различных сферах – от веб-приложений до корпоративных информационных систем. 
Фреймворк – программная платформа, задающая архитектурные принципы и общие правила разработки приложений в определенной предметной области.
Веб-браузер – прикладное программное обеспечение для поиска и просмотра веб-страниц, а также управления веб-приложениями;
Веб-приложение –это клиент-серверное программное решение, доступ к которому осуществляется через сетевой браузер посредством передачи данных по протоколу HTTP, размещается на удаленном сервере
AJAX (Asynchronous JavaScript and XML) – это веб-технология, которая позволяет обмениваться запросами с сервером, отправлять и получать от него данные без перезагрузки страницы.
.NET – это платформа для разработчиков, состоящая из инструментов, языков программирования и библиотек для создания множества различных типов приложений. ASP.NET Core расширяет платформу разработчика .NET инструментами и библиотеками, специально предназначенными для создания веб-приложений.
MVC – это архитектурный паттерн, предписывающий разделение программного кода на три взаимосвязанных компонента: модель данных, представление для визуализации информации и контроллер для обработки пользовательских запросов.

using System;
using System.Collections.Generic;
using Xunit;
using MortgageCalculatorApp;

public class MortgageCalculatorTests
{
    [Fact]
    public void CalculateLoanAmount_CorrectValue()
    {
        Assert.Equal(400000, MortgageCalculator.CalculateLoanAmount(500000, 100000));
        Assert.Equal(500000, MortgageCalculator.CalculateLoanAmount(700000, 200000));
    }

    [Fact]
    public void CalculateMonthlyPayment_CorrectValue()
    {
        Assert.Equal(4824.32, MortgageCalculator.CalculateMonthlyPayment(500000, 10, 20), 2);
        Assert.Equal(2866.12, MortgageCalculator.CalculateMonthlyPayment(300000, 8, 15), 2);
    }

    [Fact]
    public void CalculateTotalPayment_CorrectValue()
    {
        Assert.Equal(1157836.80, MortgageCalculator.CalculateTotalPayment(4824.32, 20), 2);
        Assert.Equal(515901.60, MortgageCalculator.CalculateTotalPayment(2866.12, 15), 2);
    }

    [Fact]
    public void CalculateRequired_CorrectValue()
    {
        Assert.Equal(16081.07, MortgageCalculator.CalculateRequiredIncome(4824.32, 30), 2);
        Assert.Equal(11464.48, MortgageCalculator.CalculateRequiredIncome(2866.12, 25), 2);
    }

    [Fact]
    public void GetBankOffers_ValidOffers()
    {
        var offers = MortgageCalculator.GetBankOffers(500000, 10, 20);
        Assert.NotNull(offers);
        Assert.NotEmpty(offers);
        Assert.All(offers, offer => Assert.NotNull(offer.BankName));
        Assert.All(offers, offer => Assert.True(offer.InterestRate > 0));
        Assert.All(offers, offer => Assert.True(offer.MonthlyPayment > 0));
    }
}
